520:             const CACHE_PATH_NO_SCHEME = 1; // Exclude scheme.
521:             const CACHE_PATH_NO_HOST = 2; // Exclude host (i.e. domain name).
522:             const CACHE_PATH_NO_PATH = 4; // Exclude path (i.e. the request URI).
523:             const CACHE_PATH_NO_PATH_INDEX = 8; // Exclude path index (i.e. no default `index`).
524:             const CACHE_PATH_NO_QUV = 16; // Exclude query, user & version salt.
525:             const CACHE_PATH_NO_QUERY = 32; // Exclude query string.
526:             const CACHE_PATH_NO_USER = 64; // Exclude user token.
527:             const CACHE_PATH_NO_VSALT = 128; // Exclude version salt.
528:             const CACHE_PATH_NO_EXT = 256; // Exclude extension.

530:             public function url_to_cache_path($url, $with_user_token = '', $with_version_salt = '', $flags = 0)
531:                 {
532:                     $cache_path        = ''; // Initialize.
533:                     $url               = trim((string)$url);
534:                     $with_user_token   = trim((string)$with_user_token);
535:                     $with_version_salt = trim((string)$with_version_salt);
536: 
537:                     if($url && strpos($url, '://') === FALSE)
538:                         $url = '//'.ltrim($url, '/');
539: 
540:                     if(!$url || !($url = parse_url($url)))
541:                         return ''; // Invalid URL.
542: 
543:                     if(!($flags & $this::CACHE_PATH_NO_SCHEME))
544:                         {
545:                             if(!empty($url['scheme']))
546:                                 $cache_path .= $url['scheme'].'/';
547:                             else $cache_path .= $this->is_ssl() ? 'https/' : 'http/';
548:                         }
549:                     if(!($flags & $this::CACHE_PATH_NO_HOST))
550:                         {
551:                             if(!empty($url['host']))
552:                                 $cache_path .= $url['host'].'/';
553:                             else $cache_path .= $_SERVER['HTTP_HOST'].'/';
554:                         }
555:                     if(!($flags & $this::CACHE_PATH_NO_PATH))
556:                         {
557:                             if(!empty($url['path']) && strlen($url['path'] = trim($url['path'], '\\/'." \t\n\r\0\x0B")))
558:                                 $cache_path .= $url['path'].'/';
559:                             else if(!($flags & $this::CACHE_PATH_NO_PATH_INDEX)) $cache_path .= 'index/';
560:                         }
561:                     $cache_path = str_replace('.', '-', $cache_path);
562: 
563:                     if(!($flags & $this::CACHE_PATH_NO_QUV))
564:                         {
565:                             if(!($flags & $this::CACHE_PATH_NO_QUERY))
566:                                 if(isset($url['query']) && $url['query'] !== '')
567:                                     $cache_path = rtrim($cache_path, '/').'.q/'.md5($url['query']).'/';
568: 
569:                             if(!($flags & $this::CACHE_PATH_NO_USER))
570:                                 if($with_user_token !== '') // Allow a `0` value if desirable.
571:                                     $cache_path = rtrim($cache_path, '/').'.u/'.str_replace(array('/', '\\'), '-', $with_user_token).'/';
572: 
573:                             if(!($flags & $this::CACHE_PATH_NO_VSALT))
574:                                 if($with_version_salt !== '') // Allow a `0` value if desirable.
575:                                     $cache_path = rtrim($cache_path, '/').'.v/'.str_replace(array('/', '\\'), '-', $with_version_salt).'/';
576:                         }
577:                     $cache_path = trim(preg_replace('/\/+/', '/', $cache_path), '/');
578:                     $cache_path = preg_replace('/[^a-z0-9\/.]/i', '-', $cache_path);
579: 
580:                     if(!($flags & $this::CACHE_PATH_NO_EXT))
581:                         $cache_path .= '.html';
582: 
583:                     return $cache_path;
584:                 }
